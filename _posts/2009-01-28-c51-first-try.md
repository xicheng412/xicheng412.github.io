## 我的第一个成功的流水灯

下面是代码

```c
#include<reg52.h>//=====头文件申明
#define uint unsigned int    //=====预定义

void d(uint m)    //延时
{
	uint x=m;
	while(x>1){x--;}
}

void main()    
{
	uint a=1,b=1;
	while(1)
	{
		for(b=1;b<7;b++)    //向左
		{
			a=a<<1;
			P2=~a;
			d(2000*b);        //呵呵，先快后慢的动
		} 


		for(b=1;b<7;b++)    //向右
		{
			a=a>>1;
			P2=~a;
			d(2000*b);
		}
	}
}
```
